import pygame import sys import random # Game Settings WIDTH, HEIGHT = 400, 600 GRAVITY = 0.25 BIRD_JUMP = -6 PIPE_SPEED = 3 GAP_SIZE = 150 # Initialize Pygame pygame.init() screen = pygame.display.set_mode((WIDTH, HEIGHT)) clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 32) # Colors WHITE = (255, 255, 255) SKY_BLUE = (135, 206, 235) GREEN = (34, 139, 34) # --- Bird Setup --- # Dhyaan dein: Apni image ka naam 'character.png' rakhein ya niche badal dein try: bird_img = pygame.image.load('character.png').convert_alpha() bird_img = pygame.transform.scale(bird_img, (50, 40)) # Size adjust karein except: # Agar image nahi milti toh ek yellow box ban jayega bird_img = pygame.Surface((40, 30)) bird_img.fill((255, 255, 0)) bird_rect = bird_img.get_rect(center=(100, HEIGHT//2)) bird_movement = 0 # --- Pipe Setup --- pipes = [] SPAWNPIPE = pygame.USEREVENT pygame.time.set_timer(SPAWNPIPE, 1200) def create_pipe(): random_pipe_pos = random.randint(150, 450) bottom_pipe = pygame.Rect(WIDTH, random_pipe_pos, 60, HEIGHT) top_pipe = pygame.Rect(WIDTH, random_pipe_pos - GAP_SIZE - HEIGHT, 60, HEIGHT) return bottom_pipe, top_pipe def move_pipes(pipes_list): for pipe in pipes_list: pipe.centerx -= PIPE_SPEED return [pipe for pipe in pipes_list if pipe.right > -50] def draw_pipes(pipes_list): for pipe in pipes_list: pygame.draw.rect(screen, GREEN, pipe) def check_collision(pipes_list): for pipe in pipes_list: if bird_rect.colliderect(pipe): return False if bird_rect.top <= -100 or bird_rect.bottom >= HEIGHT: return False return True # --- Main Game Loop --- score = 0 game_active = True while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE and game_active: bird_movement = 0 bird_movement += BIRD_JUMP if event.key == pygame.K_SPACE and not game_active: # Reset Game game_active = True pipes.clear() bird_rect.center = (100, HEIGHT//2) bird_movement = 0 score = 0 if event.type == SPAWNPIPE and game_active: pipes.extend(create_pipe()) # Background screen.fill(SKY_BLUE) if game_active: # Bird Physics bird_movement += GRAVITY bird_rect.centery += bird_movement screen.blit(bird_img, bird_rect) # Pipes pipes = move_pipes(pipes) draw_pipes(pipes) # Collision game_active = check_collision(pipes) # Simple Score (Pipes count / 2 kyunki top/bottom dono hain) score += 0.01 else: text = font.render(f"Game Over! Score: {int(score)}", True, WHITE) screen.blit(text, (WIDTH//2 - 100, HEIGHT//2)) retry_text = font.render("Press Space to Restart", True, WHITE) screen.blit(retry_text, (WIDTH//2 - 130, HEIGHT//2 + 50)) pygame.display.update() clock.tick(60)
